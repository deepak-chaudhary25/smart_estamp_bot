var lastValues = new WeakMap();

/* ===========================
   STRICT HUMAN INPUT TRACKING
   (Input + Textarea + Select)

function recordTyping(e) {
  if (!e.isTrusted) return;

  if (autofillLocked) return;

  var el = e.target;
  if (!el) return;
  
  issueTypingToken();

  if (el.tagName === "SELECT") {
    totalTypedChars += 1;

    if (!firstTypeTime) {
      firstTypeTime = Date.now();
      setHidden("firstTypeTs", firstTypeTime);
    }

    setHidden("typedCharCount", totalTypedChars);
    return;
  }

  if (!el.value) return;

  var prev = lastValues.get(el) || "";
  var delta = Math.abs(el.value.length - prev.length);

  if (delta > 0) {
    totalTypedChars += delta;

    if (!firstTypeTime) {
      firstTypeTime = Date.now();
      setHidden("firstTypeTs", firstTypeTime);
    }

    setHidden("typedCharCount", totalTypedChars);
    lastValues.set(el, el.value);
  }
}

function calculateSurcharge(stampDutyPayable,percent){
	
	var surcharge = (percent/100)*stampDutyPayable;
	//var surchargeRounding = roundSurcharge(surcharge);
	//CR03_12022018_SurchargeUpdate - commented above code and added below rounding normal logic
	var surchargeRounding = Math.round(surcharge);
	return surchargeRounding ;
}

function roundSurcharge(surcharge){
	
	var surchargeRound;
	var surchargeStr = surcharge.toString();
	var decimalPosition = surchargeStr.indexOf(".");
	
	//If decimal character present
    if(decimalPosition >= 0){
		var res = surchargeStr.split("."); 
	    var wholeNum = res[0];
	    var deimalNum = res[1];
	   
	    if(parseInt(wholeNum)<10){
	    	surchargeRound = 10;
	    }
	    else if(parseInt(wholeNum)%10 == 0){
	    	var firstDigit = parseInt(deimalNum.charAt(0)); 
	    	if(firstDigit>=5){
	    		surchargeRound = Math.ceil((parseInt(wholeNum)+1)/10)*10;
		    }else{
		    	surchargeRound = wholeNum;
	        }
	    }
	    else{
	    	surchargeRound = Math.ceil(parseInt(wholeNum)/10)*10;
	    }
    }
    else{
    	// else If whole number is entered
    	if(surcharge<10){
	    	surchargeRound = 10;
	    }
	    else if(surcharge%10 == 0){
		   surchargeRound = surcharge;
	    }
	    else{
	    	surchargeRound = Math.ceil(surcharge/10)*10;
	    }
    }
    return surchargeRound;
}

function replaceEnterKey(textarea,replaceWith)
{
    textarea.value = escape(textarea.value)
    for (i = 0; i < textarea.value.length; i++) {
        if (textarea.value.indexOf("%0D%0A") > -1) {
            textarea.value = textarea.value.replace("%0D%0A", replaceWith)
        } else if (textarea.value.indexOf("%0A") > -1) {
            textarea.value = textarea.value.replace("%0A", replaceWith)
        } else if (textarea.value.indexOf("%0D") > -1) {
            textarea.value = textarea.value.replace("%0D", replaceWith)
        }
    }
    textarea.value = unescape(textarea.value)
}

function chkInvalidDate(fld) {
		//if(!fnIsBlank(fld.value)) {
		if(fld.value!="" ||fld.value!=null) {
			var theString = new String(fld.value);
			var delimiterCharacter="-";
			var ok = true;
			var frmt = true;
			var leap = false;
			var isMonth = false;
			monthArr = new Array ("JAN","FEB","MAR","APR","MAY","JUN","JUL","AUG","SEP","OCT","NOV","DEC");
			dateArr = new Array (31,28,31,30,31,30,31,31,30,31,30,31);
			dateLeapArr = new Array (31,29,31,30,31,30,31,31,30,31,30,31);

			if (theString.length != 11){
				frmt = false;
			} else {
				if (theString.charAt(2) != "-" && theString.charAt(6) != "-") {
					frmt = false;
				} else {
					var tokens = theString.split(delimiterCharacter);
					if (tokens.length!=3) {
						frmt = false;
					} else {
						var date = tokens[0];
						var month = tokens[1].toUpperCase();
						var year = tokens[2];
						if (isNaN(year) || isNaN(date)) {
							ok = false;
						} else {
							if (year%4==0) {
								if (year%100==0) {
									if (year%400==0)
										leap=true;
								} else {
									leap = true;
								}
							}
							for (var i=0; i<12; i++) {
								if (month==monthArr[i]) {
									if (leap) {
										if (date>dateLeapArr[i])
											ok = false;
									} else {
										if (date>dateArr[i])
											ok = false;
									}
									isMonth = true;
									break;
								}
							}
							if (!isMonth)
								ok = false;
						}
					}
				}
			}
			if (!ok) {
				//alert("Please enter valid Date.");
				alert('Please enter valid Date.');
				fld.value = "";
				//fld.focus();
				return true;
			}
			if (!frmt) {
				//alert("Please enter the date in DD-MTH-YYYY format.");
				alert('Please enter the date in DD-MMM-YYYY format.');
				fld.value = "";
				//fld.focus();
				return true;
			}
		}
	 } // end of chkInvalidDate

function roundNumber(num, ndec, addz)
{
     if (num != "")
     {
          var factor = Math.pow(10, ndec);

          num = Math.round(num * factor);
          num = num / factor;
          if (addz)
          {
               var dot = ("" + num).indexOf(".");
               if (dot>=0)
               {
                    var nzeros = dot + ndec + 1 - ("" + num).length;
                    for (i=0; i< nzeros; i++)
                    {
                         num = num + "0";
                    }
               }
               else
               {
                    num = num + ".";
                    for (i=0; i< ndec; i++)
                    {
                         num = num + "0";
                    }
               }
          }
     }
     return (num);
}
