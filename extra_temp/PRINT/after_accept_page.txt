function openPost(url, variables){
    var form = document.createElement("form");

    form.setAttribute("action", url);
    for(variable in variables)
    {

function queryESession() {
		try {
			var printBtnTable = document.getElementById("PrintBtnTable");
			var PCPLinkTable = document.getElementById("PCPLinkTable");

			printBtnTable.style.display = "none";
			PCPLinkTable.style.display = "none";
	
			fetchESessionFromPCP();
			
		} catch (error) {
			printBtnTable.style.display = "none";
			PCPLinkTable.style.display = "block";
			alert("eSession request failed");
		}
}
function fetchESessionFromPCP(retries = 0, maxRetries = 5, delay = 2000) {
		var printBtnTable = document.getElementById("PrintBtnTable");
		var PCPLinkTable = document.getElementById("PCPLinkTable");
		PrintService.ajaxQueryESession().done(function(esession) {
	        if (esession && esession !== "") {
	            postEsession(esession);
	            printBtnTable.style.display = "table";
	            PCPLinkTable.style.display = "none";
	        } else {
	            console.warn("Print Control: Could not get e-session.");
	            if (retries < maxRetries) {
	                setTimeout(() => fetchESessionFromPCP(retries + 1, maxRetries, delay), delay);
	            } else {
	                alert("Print Control: Failed to retrieve e-session after multiple attempts.");
	                printBtnTable.style.display = "none";
	                PCPLinkTable.style.display = "block";
	            }
	        }
	    }).fail(function() {
	        console.warn("Print Control: Could not connect to Print Control.");
	        if (retries < maxRetries) {
	        	printBtnTable.style.display = "none";
	            PCPLinkTable.style.display = "block";
	            setTimeout(() => fetchESessionFromPCP(retries + 1, maxRetries, delay), delay);
	        } else {
	            alert("Print Control: Could not connect to Print Control after multiple attempts.");
	            printBtnTable.style.display = "none";
	            PCPLinkTable.style.display = "block";
	        }
	    });
}
function postEsession(esession) {
		$.ajax({
			type : "POST",
			url : "/eStampIndia/submission/SubmissionServlet",
			data : {
				rDoAction : "saveEsessionKey",
				esession : JSON.stringify(esession)
			},
			dataType : 'json',
			success : function(text) {

				if (text.response == "true") {
					saveEsession = true;
					return true;
				} else {
					alert("Failed to save eSession");
					return true;
				}
			}
		});

}
